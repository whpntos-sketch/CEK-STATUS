    tr:nth-child(even) { background: #f2f2f2; }
    .fade-in { animation: fadeIn 0.5s ease-in; }
    @keyframes fadeIn { from {opacity:0;} to {opacity:1;} }
    .pagination { margin-top: 20px; text-align: center; }
    .pagination button { margin: 2px; padding: 5px 10px; }

<body>
  <h1>Cek Status Transaksi</h1>
  <div style="text-align:center;">
    <input type="text" id="telepon" placeholder="Masukkan Nomor Telepon">
    <button onclick="cekStatus()">Cek</button>
  </div>
  <div id="result"></div>

  <script>
    const API_URL = "PASTE_URL_APPS_SCRIPT_HERE"; // ganti dengan URL Web App

    let currentPage = 1;
    const rowsPerPage = 5;
    let currentData = [];

    async function cekStatus() {
      const telepon = document.getElementById("telepon").value;
      if (!telepon) return alert("Masukkan nomor telepon!");
      const res = await fetch(`${API_URL}?telepon=${telepon}`);
      const data = await res.json();
      currentData = data;
      currentPage = 1;
      renderTable();
    }

    function renderTable() {
      if (currentData.length === 0) {
        document.getElementById("result").innerHTML = "<p>Tidak ada data ditemukan.</p>";
        return;
      }

      let table = "<table class='fade-in'><thead><tr>";
      const headers = Object.keys(currentData[0]);
      headers.forEach(h => table += `<th onclick="sortTable('${h}')">${h}</th>`);
      table += "</tr></thead><tbody>";

      const start = (currentPage - 1) * rowsPerPage;
      const end = start + rowsPerPage;
      const pageData = currentData.slice(start, end);

      pageData.forEach(row => {
        table += "<tr>";
        headers.forEach(h => table += `<td>${row[h]}</td>`);
        table += "</tr>";
      });

      table += "</tbody></table>";
      table += renderPagination();
      document.getElementById("result").innerHTML = table;
    }

    function sortTable(header) {
      currentData.sort((a, b) => (a[header] > b[header]) ? 1 : -1);
      renderTable();
    }

    function renderPagination() {
      const totalPages = Math.ceil(currentData.length / rowsPerPage);
      let buttons = "<div class='pagination'>";
      for (let i = 1; i <= totalPages; i++) {
        buttons += `<button onclick="goToPage(${i})">${i}</button>`;
      }
      buttons += "</div>";
      return buttons;
    }

    function goToPage(page) {
      currentPage = page;
      renderTable();
    }
  </script>
